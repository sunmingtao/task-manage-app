curl -X POST http://localhost:8000/api/auth/register/ -H "Content-Type: application/json" -d '{"username": "mike", "password": "0102", "email": "mike@example.com"}'

curl -X POST http://localhost:8000/api/auth/token/ -H "Content-Type: application/json" -d '{"username": "mike", "password": "0102"}'

ACCESS_TOKEN=$(curl -X POST http://localhost:8000/api/auth/token/ -H "Content-Type: application/json" -d '{"username": "mike", "password": "0102"}' | jq -r '.access')

curl -X GET http://localhost:8000/api/auth/profile/ \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $ACCESS_TOKEN"

# Create a board
curl -X POST http://localhost:8000/api/boards/   -H "Authorization: Bearer $ACCESS_TOKEN"   -H "Content-Type: application/json"   -d '{"title": "My First Board", "description": "Learning Django + React"}'

# View all boards
curl -X GET http://localhost:8000/api/boards/   -H "Authorization: Bearer $ACCESS_TOKEN"

# Create a list
curl -X POST http://localhost:8000/api/lists/   -H "Authorization: Bearer $ACCESS_TOKEN"   -H "Content-Type: application/json"   -d '{"title": "To Do", "board": 1}'

# Create a task
curl -X POST http://localhost:8000/api/tasks/   -H "Authorization: Bearer $ACCESS_TOKEN"   -H "Content-Type: application/json"   -d '{"title": "Learn Django Models", "description": "Understand how Django ORM works", "list": 1, "priority": "high"}'

# Create a task with assignee
curl -X POST http://localhost:8000/api/tasks/   -H "Authorization: Bearer $ACCESS_TOKEN"   -H "Content-Type: application/json"   -d '{"title": "Review Code", "description": "Code review task", "list": 1, "priority": "medium", "assignee_id": 2}'

# Update task assignment
curl -X PUT http://localhost:8000/api/tasks/1/   -H "Authorization: Bearer $ACCESS_TOKEN"   -H "Content-Type: application/json"   -d '{"title": "Learn Django Models", "description": "Understand how Django ORM works", "list": 1, "priority": "high", "assignee_id": 2}'

# View a board
curl -X GET http://localhost:8000/api/boards/1/   -H "Authorization: Bearer $ACCESS_TOKEN"

# Get all users for assignment dropdown (any board ID works)
curl -X GET http://localhost:8000/api/boards/1/users/   -H "Authorization: Bearer $ACCESS_TOKEN"

# Note: All boards are now visible to all authenticated users
# Tasks can be assigned to any user in the system

# Show migrations
python manage.py showmigrations
